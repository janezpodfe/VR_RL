        done = False

            done = True

            reward += 1.0

        if self._is_collision(self.agent, self.object):
            coll_normal = self.get_puck_position() - self.get_agent_position()
            coll_normal = coll_normal/np.linalg.norm(coll_normal)
            F_comp = self.calculate_component(self.get_puck_position(), self.get_goal_position(), coll_normal)
            reward += 0.05*F_comp  

        if self._is_collision(self.agent, self.border):
            reward += -1.0
            done = True
