import gym
import numpy as np
import pygame
import circ_env

env = gym.make('circ_env/Circle-v0', render_mode="human")

# Reset the environment
obs = env.reset()
env.render()

# Run the environment with random actions
#for i in range(500):

start_pos = None
reset_time = 0

while True:

    #if any([event.type == pygame.QUIT for event in pygame.event.get()]): break

    #player controls
    keys = pygame.key.get_pressed() 
    if keys[pygame.K_LEFT]: x = -1
    elif keys[pygame.K_RIGHT]: x = 1
    else: x = 0
    if keys[pygame.K_UP]: y = -1
    elif keys[pygame.K_DOWN]: y = 1
    else: y = 0   

    for event in pygame.event.get():
        if event.type == pygame.MOUSEBUTTONDOWN:
            # Set the starting position
            start_pos = pygame.mouse.get_pos()
        elif event.type == pygame.MOUSEBUTTONUP:
            # Reset the starting position
            start_pos = None

    if start_pos is not None:
        current_pos = pygame.mouse.get_pos()
        if pygame.time.get_ticks() - reset_time >= 100:
            # Reset the starting position every second
            start_pos = current_pos
            reset_time = pygame.time.get_ticks()        
        dx = current_pos[0] - start_pos[0]
        dy = current_pos[1] - start_pos[1]

        x = np.sign(dx)
        y = np.sign(dy)

    action = np.array([x,y],dtype=np.float32)

    #action = env.action_space.sample()
    #print(action)
    obs, reward, done, _ = env.step(action)

    # Render the environment
    env.render()

    # Check if the episode is finished
    if done:
        obs = env.reset()
        start_pos = None
        reset_time = 0

# Close the environment
env.close()
